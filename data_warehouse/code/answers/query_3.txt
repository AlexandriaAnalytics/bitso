## Query 3 - dentify on a given day which users have made more than 5 deposits historically
select
  event_date_id,
  user_id, 
  count(*) as total_deposits
from 
  dimensional_bitso_tx
where
  event_date_id < 20220928  -- dummy date
  and type = "deposit"
group by 
  event_date_id,
  user_id
having count(*) > 5;
